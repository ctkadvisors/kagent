    ```typescript
    /**
     * SPDX-License-Identifier: MIT
     * Copyright (c) 2026 Chris Knuteson
     */
    /**
     * narrowByDispositionOverlay — Phase 1 / DISP-02. The AgentDisposition
     * overlay's proposalScope.mayProposeAgainst narrows the existing
     * capability-JWT scope at proposal-issuance time. NEVER widens.
     *
     * Self-proposal, never self-promotion (D6): the overlay states which
     * proposal kinds the agent MAY propose against; tools mapped to
     * proposal kinds NOT in mayProposeAgainst are removed from
     * `claims.tools`. The existing claimsSubsetViolations defense-in-depth
     * check after parent-narrowing is unaffected — narrowing produces a
     * subset of the input claims, so subset invariants are preserved.
     *
     * Pure function. Caller (cap-issuer.ts) is responsible for emitting
     * disposition.proposal_rejected audit events from the returned
     * rejections list.
     */
    import type { CapabilityClaims } from '@kagent/capability-types';
    import { classifyToolAsProposal, type ProposalKind } from './proposal-tool-map.js';
    import type { DispositionOverlay } from './overlay-loader.js';

    export interface ProposalRejection {
      readonly tool: string;
      readonly kind: ProposalKind;
      readonly agentRef: string;
      readonly agentNamespace: string;
      readonly agentName: string;
      readonly dispositionConfigMapName: string;
      readonly dispositionConfigMapNamespace: string;
      readonly mayProposeAgainst: readonly ProposalKind[];
      readonly reason: 'not_in_mayProposeAgainst';
    }

    export interface NarrowResult {
      readonly narrowed: CapabilityClaims;
      readonly rejections: readonly ProposalRejection[];
    }

    export function narrowByDispositionOverlay(
      claims: CapabilityClaims,
      overlay: DispositionOverlay | null,
    ): NarrowResult {
      if (overlay === null) return { narrowed: claims, rejections: [] };
      if (claims.tools === undefined) return { narrowed: claims, rejections: [] };

      const allow = new Set<ProposalKind>(overlay.idleBehavior.proposalScope.mayProposeAgainst);
      const keptTools: string[] = [];
      const rejections: ProposalRejection[] = [];

      for (const tool of claims.tools) {
        const kind = classifyToolAsProposal(tool);
        if (kind === null) {
          keptTools.push(tool);
          continue;
        }
        if (allow.has(kind)) {
          keptTools.push(tool);
          continue;
        }
        rejections.push({
          tool,
          kind,
          agentRef: overlay.agentRef,
          agentNamespace: overlay.agentNamespace,
          agentName: overlay.agentName,
          dispositionConfigMapName: overlay.configMapName,
          dispositionConfigMapNamespace: overlay.configMapNamespace,
          mayProposeAgainst: overlay.idleBehavior.proposalScope.mayProposeAgainst,
          reason: 'not_in_mayProposeAgainst',
        });
      }

      // Build narrowed claims by spreading the input and replacing only `tools`.
      // This guarantees other claim categories (read/write/spawn/models/egress/tenant/...)
      // pass through unchanged.
      const narrowed: CapabilityClaims = { ...claims, tools: Object.freeze(keptTools) };
      return { narrowed, rejections: Object.freeze(rejections) };
    }
    ```

    ```typescript
     * proposals-counter — Phase 1 / DISP-02 + DISP-03 bridge. The operator's
     * cap-issuer narrowing step calls incrementProposalsToday() when a
     * proposal-claim mint succeeds for an Agent with a disposition overlay.
     * The function PATCHes the disposition ConfigMap's annotations:

     *   kagent.knuteson.io/proposals-today-day  ← current UTC day (YYYY-MM-DD)

     * dispositions projection (plan 03) reads to compute proposalsToday —
     * workbench-api never writes this annotation.
     *
     * Daily rollover semantics: if the existing day annotation mismatches
     * today's UTC day, the increment STARTS at 1 (the previous day's
     * count is implicitly reset).
     *
     * No new persistence primitive: ConfigMap annotations are an existing
     * Kubernetes primitive; the configmaps:patch RBAC was granted in plan 01.
     */
    import type { CoreV1Api } from '@kubernetes/client-node';
    import {
      DISPOSITION_PROPOSALS_TODAY_ANNOTATION,
      DISPOSITION_PROPOSALS_TODAY_DAY_ANNOTATION,
      type DispositionOverlay,
    } from '@kagent/dto';

    /** Pure helper. No I/O. Tested independently. */
    export function computeNextProposalsTodayPatch(args: {
      readonly currentValue: string | undefined;
      readonly currentDay: string | undefined;
      readonly todayDay: string;          // YYYY-MM-DD UTC
    }): { readonly nextValue: string; readonly nextDay: string } {
      const dayMatches = args.currentDay !== undefined && args.currentDay === args.todayDay;
      const parsed = args.currentValue !== undefined && args.currentValue !== ''
        ? Math.floor(Number(args.currentValue))
        : NaN;
      const baseline = dayMatches && Number.isFinite(parsed) && parsed >= 0 ? parsed : 0;
      return { nextValue: String(baseline + 1), nextDay: args.todayDay };
    }

    /** Format `now` as YYYY-MM-DD UTC. */
    export function formatUtcDay(now: Date): string {
      const y = now.getUTCFullYear();
      const m = String(now.getUTCMonth() + 1).padStart(2, '0');
      const d = String(now.getUTCDate()).padStart(2, '0');
      return `${y}-${m}-${d}`;
    }

    const PROPOSALS_TODAY_MAX_RETRIES = 3;

    /** Detect HTTP 409 Conflict from @kubernetes/client-node v1.x error shape. */
    function isConflict(err: unknown): boolean {
      const e = err as { response?: { statusCode?: number }; statusCode?: number; code?: number };
      const status = e?.response?.statusCode ?? e?.statusCode ?? e?.code;
      return status === 409;
    }

    export async function incrementProposalsToday(args: {
      readonly coreApi: Pick<CoreV1Api, 'patchNamespacedConfigMap' | 'readNamespacedConfigMap'>;
      readonly overlay: DispositionOverlay;
      readonly now: Date;
      readonly logger?: { warn(msg: string): void; info?(msg: string): void };
    }): Promise<void> {
      const log = args.logger ?? { warn: (m: string) => console.warn(m), info: (m: string) => console.debug(m) };
      const todayDay = formatUtcDay(args.now);

      // Optimistic-concurrency CAS loop. The counter is the SOLE writer of these
      // two annotations BUT multiple operator goroutines may race on parallel
      // mints for the same Agent — without CAS, one increment overwrites another.
      // Per Codex MEDIUM #4: capture metadata.resourceVersion on read; issue a
      // JSON-Patch test+replace; on 409 conflict, re-read and retry up to 3 times;
      // on give-up, log warn and return (counter is best-effort observation).
      for (let attempt = 0; attempt <= PROPOSALS_TODAY_MAX_RETRIES; attempt++) {
        try {
          const cm = await args.coreApi.readNamespacedConfigMap(
            args.overlay.configMapName,
            args.overlay.configMapNamespace,
          );
          const cmBody = (cm as { body?: { metadata?: { annotations?: Record<string, string>; resourceVersion?: string } }; metadata?: { annotations?: Record<string, string>; resourceVersion?: string } }).body
            ?? (cm as { metadata?: { annotations?: Record<string, string>; resourceVersion?: string } });
          const ann = cmBody?.metadata?.annotations ?? {};
          const resourceVersion = cmBody?.metadata?.resourceVersion;
          if (resourceVersion === undefined) {
            log.warn('proposals-counter: missing resourceVersion for ' + args.overlay.agentRef + '; skipping CAS');
            return;
          }
          const next = computeNextProposalsTodayPatch({
            currentValue: ann[DISPOSITION_PROPOSALS_TODAY_ANNOTATION],
            currentDay: ann[DISPOSITION_PROPOSALS_TODAY_DAY_ANNOTATION],
            todayDay,
          });
          // JSON-Patch test+replace: the `test` op fails with 409 Conflict if
          // the resourceVersion no longer matches because a parallel goroutine
          // patched first. Annotation keys use ~1 to escape `/` in JSON-Pointer.
          const patchBody = [
            { op: 'test', path: '/metadata/resourceVersion', value: resourceVersion },
            { op: 'replace', path: '/metadata/annotations/kagent.knuteson.io~1proposals-today', value: next.nextValue },
            { op: 'replace', path: '/metadata/annotations/kagent.knuteson.io~1proposals-today-day', value: next.nextDay },
          ] as const;
          await args.coreApi.patchNamespacedConfigMap(
            args.overlay.configMapName,
            args.overlay.configMapNamespace,
            patchBody as unknown as object,
            // The @kubernetes/client-node v1.x patch call accepts an options
            // object — set Content-Type to `application/json-patch+json` via the
            // headers option. Mirror whatever pattern the operator codebase
            // already uses; search via `grep -rn "patchNamespacedConfigMap" packages/operator/src/`.
            // If no precedent exists, pass `{ headers: { 'Content-Type': 'application/json-patch+json' } }`.
          );
          return; // success
        } catch (e) {
          if (isConflict(e) && attempt < PROPOSALS_TODAY_MAX_RETRIES) {
            log.info?.('proposals-counter: 409 conflict for ' + args.overlay.agentRef + ' attempt=' + (attempt + 1) + '; retrying');
            continue;

          if (isConflict(e)) {
            log.warn('proposals-counter: gave up after ' + PROPOSALS_TODAY_MAX_RETRIES + ' conflicts for ' + args.overlay.agentRef);
            return;
          }
          log.warn('proposals-counter: increment failed for ' + args.overlay.agentRef + ': ' + (e instanceof Error ? e.message : String(e)));
          return;
        }
      }
    }
    ```
