     * DispositionOverlayRow — Phase 1 / DISP-03 read projection.
     *
     * The workbench-api computes this row per Agent that has an
     * AgentDisposition overlay (sibling ConfigMap labeled
     * `kagent.knuteson.io/agent-disposition=true`). Spec fields mirror
     * the overlay's `data.disposition.yaml`; counter fields are computed
     * projections from existing telemetry (gateway token-usage + operator-

     * persistence primitive — D2.
     */

    export type DispositionOverBudgetReason = 'tokens_exceeded' | 'proposals_exceeded' | 'both';

    export interface DispositionOverlayRow {
      /** "namespace/name" of the Agent. */
      readonly agentRef: string;
      readonly namespace: string;
      readonly agentName: string;
      readonly configMapName: string;
      /** Spec fields from the ConfigMap's `data.disposition.yaml`. */
      readonly idleBehavior: {
        readonly readChannels: readonly string[];
        readonly attentionBudget: {
          readonly tokensPerDay: number;
          readonly pollIntervalSeconds: number;
        };
        readonly proposalScope: {
          readonly mayProposeAgainst: readonly DispositionProposalKind[];
          readonly maxProposalsPerDay: number;
        };
      };
      /**
       * Sum of `inputTokens + outputTokens` across gateway usage rows
       * with this Agent's name and `occurredAt >= dailyBoundaryUtc`.
       */
      readonly spentTokensToday: number;
      /**
       * Always 0 in v0.2. Posts/Channels are Future Research; this field
       * is reserved for forward compatibility and is set to a literal `0`.
       */
      readonly postsToday: 0;
      /**
       * Count of proposal-scope rejections for this Agent in the current
       * UTC day window. Read from the disposition ConfigMap's
       * `kagent.knuteson.io/proposals-today` annotation, written by the
       * operator's cap-issuer narrowing step.
       */

      /** Set true when spentTokensToday > tokensPerDay OR proposalsToday > maxProposalsPerDay. */
      readonly overBudget: boolean;
      /** Reason flag — undefined when overBudget is false. */
      readonly overBudgetReason?: DispositionOverBudgetReason;
      /** ISO 8601 timestamp of the start of the current day window (UTC midnight by default). */
      readonly dailyBoundaryUtc: string;
    }

    /**
     * Runtime shape check — used by workbench-ui to fail fast if the API
     * payload changes. Throws on mismatch.
     */
    export function assertIsDispositionOverlayRow(value: unknown): asserts value is DispositionOverlayRow {
      if (typeof value !== 'object' || value === null) throw new Error('not an object');
      const r = value as Record<string, unknown>;
      if (typeof r['agentRef'] !== 'string') throw new Error('agentRef missing');
      if (typeof r['namespace'] !== 'string') throw new Error('namespace missing');
      if (typeof r['agentName'] !== 'string') throw new Error('agentName missing');
      if (typeof r['configMapName'] !== 'string') throw new Error('configMapName missing');
      if (typeof r['spentTokensToday'] !== 'number') throw new Error('spentTokensToday missing');
      if (r['postsToday'] !== 0) throw new Error('postsToday must be 0 in v0.2');
      if (typeof r['proposalsToday'] !== 'number') throw new Error('proposalsToday missing');
      if (typeof r['overBudget'] !== 'boolean') throw new Error('overBudget missing');
      if (typeof r['dailyBoundaryUtc'] !== 'string') throw new Error('dailyBoundaryUtc missing');
      if (typeof r['idleBehavior'] !== 'object' || r['idleBehavior'] === null) throw new Error('idleBehavior missing');
      // ... (do not exhaustively validate sub-objects; the workbench-api
      // produces the row with full type-coverage so this check is
      // primarily a UI-side defense against schema drift.)
    }

     * GET /api/dispositions — Phase 1 / DISP-03 read projection.
     *
     * Computes per-Agent disposition rows from existing telemetry:
     *   - Spec fields ← sibling ConfigMap labeled
     *     `kagent.knuteson.io/agent-disposition=true` (annotation
     *     `kagent.knuteson.io/agent-ref=<ns>/<name>` ties to Agent).
     *   - spentTokensToday ← sum of (inputTokens + outputTokens) across
     *     GatewayUsageRow filtered by agentName + occurredAt >=
     *     dailyBoundaryUtc.
     *   - proposalsToday ← parseInt of ConfigMap annotation
     *     `kagent.knuteson.io/proposals-today` (operator-written;
     *     defaults to 0).
     *   - postsToday ← always 0 in v0.2 (Posts/Channels are Future
     *     Research).
     *
     * Over-budget emission: when overBudget is true, one
     * `disposition.over_budget` event per reason is emitted PER
     * (agentRef, reason) PER UTC-day. In-process de-dup map
     * (acceptable for v0.2 observation phase per CONTEXT.md research
     * Q11 #5; restart loses state but the observation window is small).
     */

    import { Hono } from 'hono';
    import type { CoreV1Api, CustomObjectsApi, V1ConfigMap } from '@kubernetes/client-node';
    import {
      parseDispositionConfigMap,
      DISPOSITION_LABEL,
      DISPOSITION_PROPOSALS_TODAY_ANNOTATION,
      DISPOSITION_PROPOSALS_TODAY_DAY_ANNOTATION,
      type DispositionOverlay,
      type DispositionOverlayRow,
    } from '@kagent/dto';
    import type { GatewayClient } from '../gateway-client.js';
    import { makeEvent, type AuditEvent } from '@kagent/audit-events';

    export interface DispositionsRouteDeps {
      readonly coreApi?: Pick<CoreV1Api, 'listConfigMapForAllNamespaces' | 'listNamespacedConfigMap'>;
      readonly readCustomApi?: Pick<CustomObjectsApi, 'getNamespacedCustomObject'>;
      readonly gatewayClient?: GatewayClient;
      readonly auditPublisher?: { publish(event: AuditEvent): Promise<void> };
      readonly now?: () => Date;
      readonly watchNamespaces?: readonly string[];
      readonly logger?: { warn(msg: string): void };
      /** In-process de-dup map; injectable for tests. Keys are `${agentRef}|${reason}|${dailyBoundaryUtc}`. */
      readonly overBudgetDedup?: Set<string>;
    }

    export function dispositionsRoute(deps: DispositionsRouteDeps): Hono {
      const app = new Hono();
      const dedup = deps.overBudgetDedup ?? new Set<string>();
      const now = deps.now ?? (() => new Date());
      const logger = deps.logger ?? { warn: (m) => console.warn(m) };

      app.get('/', async (c) => {
        if (deps.coreApi === undefined) return c.json({ items: [] satisfies DispositionOverlayRow[] });

        const today = now();
        const dailyBoundaryUtc = new Date(Date.UTC(
          today.getUTCFullYear(), today.getUTCMonth(), today.getUTCDate(),
          0, 0, 0, 0,
        )).toISOString();

        // List ConfigMaps with the disposition label across watch namespaces (or cluster-wide).
        const namespaces = deps.watchNamespaces ?? [];
        const cms: V1ConfigMap[] = [];
        if (namespaces.length === 0) {
          // Cluster-wide listing path. The @kubernetes/client-node v1.x signature is:
          //   listConfigMapForAllNamespaces(allowWatchBookmarks?, _continue?, fieldSelector?, labelSelector?, limit?, ...)
          const resp = await deps.coreApi.listConfigMapForAllNamespaces?.(
            undefined, undefined, undefined, `${DISPOSITION_LABEL}=true`,
          );
          if (resp !== undefined) cms.push(...((resp.body?.items ?? resp.items ?? []) as V1ConfigMap[]));
        } else {
          for (const ns of namespaces) {
            const resp = await deps.coreApi.listNamespacedConfigMap?.(
              ns, undefined, undefined, undefined, undefined, `${DISPOSITION_LABEL}=true`,
            );
            if (resp !== undefined) cms.push(...((resp.body?.items ?? resp.items ?? []) as V1ConfigMap[]));
          }
        }

        const items: DispositionOverlayRow[] = [];
        for (const cm of cms) {
          const parsed = parseDispositionConfigMap(cm);
          if (!parsed.ok) {
            logger.warn(`dispositions: skipping malformed ConfigMap ${cm.metadata?.namespace ?? '?'}/${cm.metadata?.name ?? '?'}: ${parsed.error}`);
            continue;
          }
          const overlay = parsed.overlay;
          const exists = await agentExists(deps, overlay.agentNamespace, overlay.agentName, logger);
          if (!exists) {
            continue;
          }

          // proposalsToday from annotation (BLOCKER #2 resolution: annotation-writer pattern,
          // per CONTEXT.md "Counter projection (no new persistence)" + RESEARCH.md Q11). The
          // operator's cap-issuer narrowing step is the SOLE writer of this annotation (plan 02);
          // workbench-api READS it here and never writes. The annotation is per-(agent,day);
          // a sibling annotation `kagent.knuteson.io/proposals-today-day` records the UTC day
          // window the count belongs to. If the day annotation is absent OR mismatches today's
          // UTC day, treat the current value as 0 (rollover semantics).
          const dayString = `${today.getUTCFullYear()}-${String(today.getUTCMonth() + 1).padStart(2, '0')}-${String(today.getUTCDate()).padStart(2, '0')}`;
          const annDayValue = cm.metadata?.annotations?.[DISPOSITION_PROPOSALS_TODAY_DAY_ANNOTATION];
          const annValue = cm.metadata?.annotations?.[DISPOSITION_PROPOSALS_TODAY_ANNOTATION];
          const dayMatches = annDayValue !== undefined && annDayValue === dayString;
          const parsedProposals = annValue === undefined || annValue === '' ? NaN : Number(annValue);
          const proposalsToday = dayMatches && annValue !== undefined && annValue !== ''
            ? (Number.isFinite(parsedProposals) ? Math.max(0, Math.floor(parsedProposals)) : 0)
            : 0;

          // spentTokensToday from gateway
          let spentTokensToday = 0;
          if (deps.gatewayClient !== undefined) {
            const usageRows = await deps.gatewayClient.usage({
              agentName: overlay.agentName,
              since: dailyBoundaryUtc,
              limit: 1000,
            });
            for (const row of usageRows) {
              spentTokensToday += (row.inputTokens ?? 0) + (row.outputTokens ?? 0);
            }
          }

          const tokensExceeded = spentTokensToday > overlay.idleBehavior.attentionBudget.tokensPerDay;
          const proposalsExceeded = proposalsToday > overlay.idleBehavior.proposalScope.maxProposalsPerDay;
          const overBudget = tokensExceeded || proposalsExceeded;
          const overBudgetReason: 'tokens_exceeded' | 'proposals_exceeded' | 'both' | undefined =
            tokensExceeded && proposalsExceeded ? 'both' :
            tokensExceeded ? 'tokens_exceeded' :
            proposalsExceeded ? 'proposals_exceeded' :
            undefined;

          // Emit over-budget audit event(s) — exactly once per (agentRef, reason) per day window
          if (overBudget && deps.auditPublisher !== undefined) {
            const reasons: ('tokens_exceeded' | 'proposals_exceeded')[] = [];
            if (tokensExceeded) reasons.push('tokens_exceeded');
            if (proposalsExceeded) reasons.push('proposals_exceeded');
            for (const reason of reasons) {
              const dedupKey = `${overlay.agentRef}|${reason}|${dailyBoundaryUtc}`;
              if (dedup.has(dedupKey)) continue;
              dedup.add(dedupKey);
              try {
                await deps.auditPublisher.publish(makeEvent({
                  type: 'disposition.over_budget',
                  source: 'kagent.knuteson.io/workbench-api',
                  subject: 'agent/' + overlay.agentName,
                  data: {
                    agentRef: overlay.agentRef,
                    agentNamespace: overlay.agentNamespace,
                    agentName: overlay.agentName,
                    dispositionConfigMapName: overlay.configMapName,
                    reason,
                    observed: reason === 'tokens_exceeded' ? spentTokensToday : proposalsToday,
                    budget: reason === 'tokens_exceeded' ? overlay.idleBehavior.attentionBudget.tokensPerDay : overlay.idleBehavior.proposalScope.maxProposalsPerDay,
                    dailyBoundaryUtc,
                  },
                }));
              } catch (e) {
                logger.warn('dispositions: over_budget publish failed: ' + (e instanceof Error ? e.message : String(e)));
                // Do NOT remove from dedup — exactly-once-or-zero is acceptable per CONTEXT.md.
                // (CONTEXT specifies "at most once per day per (agent, kind) to avoid log floods".)
              }
            }
          }

          items.push({
            agentRef: overlay.agentRef,
            namespace: overlay.agentNamespace,
            agentName: overlay.agentName,
            configMapName: overlay.configMapName,
            idleBehavior: overlay.idleBehavior,
            spentTokensToday,
            postsToday: 0,
            proposalsToday,
            overBudget,
            ...(overBudgetReason !== undefined && { overBudgetReason }),
            dailyBoundaryUtc,
          });
        }

        return c.json({ items });
      });

      return app;
    }

    async function agentExists(
      deps: Pick<DispositionsRouteDeps, 'readCustomApi'>,
      namespace: string,
      name: string,
      logger: { warn(message: string): void },
    ): Promise<boolean> {
      if (deps.readCustomApi === undefined) {
        logger.warn('dispositions: readCustomApi unavailable; cannot verify Agent existence for ' + namespace + '/' + name);
        return false;
      }
      try {
        await deps.readCustomApi.getNamespacedCustomObject({
          group: 'kagent.knuteson.io',
          version: 'v1alpha1',
          namespace,
          plural: 'agents',
          name,
        });
        return true;
      } catch (e) {
        logger.warn('dispositions: skipping orphan overlay for missing Agent ' + namespace + '/' + name + ': ' + (e instanceof Error ? e.message : String(e)));
        return false;
      }
    }
